CREATE VIEW `deppo_order_detail_list_left_view` AS
SELECT `dos`.name AS `status`, `o`.`id` AS `order_id`, `o`.`number` AS `number`, `o`.`date` AS `date`, `o`.`description` AS `description`, `scl`.`code` AS `stock_code`, `scl`.`name` AS `stock_name`, `od`.`ordered_quantity` AS `ordered_quantity`, `od`.`received_quantity` AS `received_quantity`, `od`.`description` AS `detail_description` FROM ((`deppo_order` `o` left join `deppo_order_detail` `od` on(`o`.`id` = `od`.`order_id`)) left join `deppo_stock_card_list` `scl` on(`od`.`stock_id` = `scl`.`id`)) left join `deppo_order_status` `dos` on(`o`.`status_id`=`dos`.`id`) WHERE (`od`.`ordered_quantity` > `od`.`received_quantity`) AND (`o`.`status_id`=2);
